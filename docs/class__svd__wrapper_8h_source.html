145 class_SVD::schmidt(const Eigen::Tensor<Scalar,tensor_rank> &tensor, long dL, long dR, long chiL, long chi_max, long chiR) {
146  if (dL*chiL * dR*chiR != tensor.size()){throw std::range_error("schmidt error: tensor size does not match given dimensions.");}
147  Eigen::Map<const MatrixType<Scalar>> mat (tensor.data(), dL*chiL, dR*chiR);
