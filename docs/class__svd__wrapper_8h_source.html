47 typename class_SVD<Scalar>:: template TensorTuple<2,1,2> class_SVD<Scalar>::decompose(const Textra::Tensor<Scalar,2> &tensor) {
48  Eigen::Map<const MatrixType> mat (tensor.data(), tensor.dimension(0), tensor.dimension(1));
49  SVD.compute(mat, Eigen::ComputeThinU | Eigen::ComputeThinV);
50  truncation_error = SVD.singularValues().tail(SVD.nonzeroSingularValues()-chi).sum();
51  return std::make_tuple

55  Textra::Matrix_to_Tensor<Scalar,2>(SVD.matrixV().transpose(), {SVD.matrixV().cols(), SVD.matrixV().rows() })

62 typename class_SVD<Scalar>:: template TensorTuple<2,1,2> class_SVD<Scalar>::decompose(const Textra::Tensor<Scalar,2> &tensor, const long chi_max) {
63  Eigen::Map<const MatrixType> mat (tensor.data(), tensor.dimension(0), tensor.dimension(1));
64  SVD.compute(mat, Eigen::ComputeThinU | Eigen::ComputeThinV);
65  truncation_error = SVD.singularValues().tail(SVD.nonzeroSingularValues()-chi).sum();
66  long chi = std::min(SVD.rank(), chi_max);
67  return std::make_tuple
68  (
69  Textra::Matrix_to_Tensor<Scalar,2>(SVD.matrixU().leftCols(chi), {SVD.matrixU().rows(), chi}),
70  Textra::Matrix_to_Tensor<Scalar,1>(SVD.singularValues().head(chi).normalized() , {chi}),
71  Textra::Matrix_to_Tensor<Scalar,2>(SVD.matrixV().leftCols(chi).transpose() , {chi, SVD.matrixV().rows() })
72  );
73 }

81  return std::make_tuple
