65  template <typename Scalar>
66  void subtract_phase(std::vector<Scalar> &eigvecs,int L, int nev)
67  // The solution to the eigenvalue equation Av = l*v is determined up to a constant phase factor, i.e., if v
68  // is a solution, so is v*exp(i*theta). By computing the complex angle of the first element in v, one can then
69  // remove it from all other elements of v.
70  {
71  if constexpr (std::is_same<Scalar, std::complex<double>>::value) {
72  if (nev > 0){
73  using namespace std::complex_literals;
74  for (int i = 0; i < nev; i++) {
75  Scalar inv_phase = -1.0i * std::arg(eigvecs[i * L]);
76  auto begin = eigvecs.begin() + i * L;
77  auto end = begin + L;
78  Scalar exp_inv_phase = std::exp(inv_phase);
79  std::transform(begin, end, begin,
80  [exp_inv_phase](std::complex<double> num) -> std::complex<double>
81  { return (num * exp_inv_phase); });
82  }
83  }else{
84  std::cerr << "Eigenvalues haven't been computed yet. Can't subtract phase. Exiting " << std::endl;
85  exit(1);
86  }
87  }
88  }

91  template <eigutils::eigSetting::Type type,
92  eigutils::eigSetting::Form form,
93  eigutils::eigSetting::Side side = eigutils::eigSetting::Side::R,
94  typename Derived>
95  void copy_solution(Derived &solver){
96  using namespace eigutils::eigSetting;
97  auto & eigvecs = solution.get_eigvecs<type,form,side>();
98  auto & eigvals = solution.get_eigvals<form>();
99  int eigvecsize = solution.meta.rows * solution.meta.cols;
100  int eigvalsize = solution.meta.cols;
101 
102  // Copy eigenvalues
103  eigvals.resize(eigvalsize);
104  if constexpr(form == Form::SYMMETRIC){
105  if constexpr (type == Type::REAL){
106  std::copy(solver.RawEigenvalues() ,solver.RawEigenvalues() + eigvalsize,eigvals.begin());
107  }else {
108  for (int j = 0; j < solution.meta.cols; j++) {
109 // assert(std::abs(solver.Eigenvalue(j).imag()) < 1e-15 and "Discarding imaginary part!" );
110  if(std::abs(solver.Eigenvalue(j).imag()) > solverConf.eigThreshold){
111  std::cerr << "WARNING: Discarding imaginary part: " << solver.Eigenvalue(j).imag() << std::endl;
112  }
113  eigvals[j] = solver.Eigenvalue(j).real();
114  }
115  }
116  }else if constexpr(form == Form::NONSYMMETRIC){
117  if constexpr(type == Type::REAL){
118  for (int j = 0; j < solution.meta.cols; j++) {
119  eigvals[j] = std::complex<double>(solver.EigenvalueReal(j), solver.EigenvalueImag(j));
120  }
121  }
122  else if constexpr(type == Type::CPLX){
123  for (int j = 0; j < solution.meta.cols; j++) {
124  eigvals[j] = solver.Eigenvalue(j);
125  }
126  }
127  }
128 
129 
130 
131  // Copy eigenvectors
132  if (solverConf.compute_eigvecs){
133  eigvecs.resize(eigvecsize);
134  if constexpr (type == Type::REAL and form == Form::SYMMETRIC){
135  std::copy(solver.RawEigenvectors(),solver.RawEigenvectors() + eigvecsize, eigvecs.begin());
136  }else{
137  int count = 0;
138  for (int j = 0; j < solution.meta.cols; j++){
139  for (int i = 0; i < solution.meta.rows; i++){
140  eigvecs[count++] = solver.Eigenvector(j,i);
141  }
142  }
143  }
144  }
145 
146  if(solverConf.remove_phase){
147  subtract_phase(eigvecs,solution.meta.rows,solution.meta.cols);
148  }
