180  template<typename Derived>
181  VectorType<typename Derived::Scalar> get_vH2 (const Eigen::MatrixBase<Derived> &v, const MultiComponents<typename Derived::Scalar> &multiComponents){
182  using Scalar = typename Derived::Scalar;
183  t_vH2->tic();
184  size_t log2chiL = std::log2(multiComponents.dsizes[1]);
185  size_t log2chiR = std::log2(multiComponents.dsizes[2]);
186  size_t log2spin = std::log2(multiComponents.dsizes[0]);
187  Eigen::Tensor<Scalar,3> vH2;
188  if (log2spin > log2chiL + log2chiR){
189  if (log2chiL > log2chiR){
190  Eigen::Tensor<Scalar,3> theta = Eigen::TensorMap<const Eigen::Tensor<const Scalar,3>>(v.derived().data(), multiComponents.dsizes).shuffle(Textra::array3{1,0,2});
191  vH2 =
192  theta
193  .contract(multiComponents.env2L, Textra::idx({0}, {0}))
194  .contract(multiComponents.mpo , Textra::idx({0,3}, {2,0}))
195  .contract(multiComponents.env2R, Textra::idx({0,3}, {0,2}))
196  .contract(multiComponents.mpo , Textra::idx({2,1,4}, {2,0,1}))
197  .shuffle(Textra::array3{2,0,1});
198  }
199 
200  else{
201  Eigen::Tensor<Scalar,3> theta = Eigen::TensorMap<const Eigen::Tensor<const Scalar,3>>(v.derived().data(), multiComponents.dsizes).shuffle(Textra::array3{2,0,1});
202  vH2 =
203  theta
204  .contract(multiComponents.env2R, Textra::idx({0}, {0}))
205  .contract(multiComponents.mpo , Textra::idx({0,3}, {2,1}))
206  .contract(multiComponents.env2L, Textra::idx({0,3}, {0,2}))
207  .contract(multiComponents.mpo , Textra::idx({2,4,1}, {2,0,1}))
208  .shuffle(Textra::array3{2,1,0});
209  }
210 
211  }else{
212  Eigen::Tensor<Scalar,3> theta = Eigen::TensorMap<const Eigen::Tensor<const Scalar,3>>(v.derived().data(), multiComponents.dsizes).shuffle(Textra::array3{1,0,2});
213  vH2 =
214  theta
215  .contract(multiComponents.env2L, Textra::idx({0}, {0}))
216  .contract(multiComponents.mpo , Textra::idx({0,3}, {2,0}))
217  .contract(multiComponents.mpo , Textra::idx({4,2}, {2,0}))
218  .contract(multiComponents.env2R, Textra::idx({0,2,3}, {0,2,3}))
219  .shuffle(Textra::array3{1, 0, 2});
220  }
221  t_vH2->toc();
222  return Eigen::Map<VectorType<Scalar>>(vH2.data(),vH2.size());
223  }
