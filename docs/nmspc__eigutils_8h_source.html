79  {
80  using namespace eigSetting;

83  ritz==Ritz::SA or
84  ritz==Ritz::BE
85  )
86  {
87  std::cerr << "WARNING: Invalid ritz for nonsym problem: " << RitzToString.at(ritz) << std::endl;
88  if (ritz==Ritz::LA){ritz = Ritz::LR;}
89  if (ritz==Ritz::SA){ritz = Ritz::SR;}
90  if (ritz==Ritz::BE){ritz = Ritz::LM;}
91  std::cerr << " Changed ritz to : " << RitzToString.at(ritz)<< std::endl;
92  }
93  }else if (type==Type::REAL and form==Form::SYMMETRIC) {
94  if (ritz==Ritz::LR or
