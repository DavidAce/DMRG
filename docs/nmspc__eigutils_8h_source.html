65  {
66  using namespace eigSetting;

68  if (ritz==Ritz::LA or
69  ritz==Ritz::SA or
70  ritz==Ritz::BE
71  )
72  {
73  std::cerr << "WARNING: Invalid ritz for nonsym problem: " << RitzToString.at(ritz) << std::endl;
74  if (ritz==Ritz::LA){ritz = Ritz::LR;}
75  if (ritz==Ritz::SA){ritz = Ritz::SR;}
76  if (ritz==Ritz::BE){ritz = Ritz::LM;}
77  std::cerr << " Changed ritz to : " << RitzToString.at(ritz)<< std::endl;
78  }
79  }else if (type==Type::REAL and form==Form::SYMMETRIC) {
80  if (ritz==Ritz::LR or
81  ritz==Ritz::SR or
82  ritz==Ritz::LI or
