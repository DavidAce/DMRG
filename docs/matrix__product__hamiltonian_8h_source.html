77 void DenseHamiltonianProduct<T>::MultAx(T* theta_in_, T* theta_out_) {
78  t_mul.tic();
79  Eigen::TensorMap<Eigen::Tensor<const T, 3>> Lblock_map(Lblock,shape_theta4[1], shape_theta4[1], shape_mpo4[0] );
80  Eigen::TensorMap<Eigen::Tensor<const T, 3>> Rblock_map(Rblock,shape_theta4[3], shape_theta4[3], shape_mpo4[1] );
81  Eigen::TensorMap<Eigen::Tensor<const T, 4>> HA_map (HA, shape_mpo4);
82  Eigen::TensorMap<Eigen::Tensor<const T, 4>> HB_map (HB, shape_mpo4);
83  Eigen::TensorMap<Eigen::Tensor<const T, 4>> theta_in (theta_in_, shape_theta4);
84  Eigen::TensorMap<Eigen::Tensor<T, 4>> theta_out (theta_out_, shape_theta4);
85 
86 
87 
88  //Best yet! The sparcity of the effective hamiltonian (Lblock HA HB Rblock) is about 58% nonzeros.
89  //L have shown this to be the fastest contraction ordering
90  theta_out = Lblock_map
91  .contract(theta_in, Textra::idx({0},{1}))
92  .contract(HA_map , Textra::idx({1,2},{0,2}))// idx({1,2,3},{0,4,5}))
93  .contract(HB_map , Textra::idx({3,1},{0,2}))// idx({1,2,3},{0,4,5}))
94  .contract(Rblock_map, Textra::idx({1,3},{0,2}))
95  .shuffle(Textra::array4{1,0,2,3});
96  counter++;
97  t_mul.toc();
98 }
