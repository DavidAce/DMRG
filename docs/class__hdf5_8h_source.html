176  template<typename DataType>
177  hid_t get_DataType(){
178  if(std::is_same<DataType, int>::value) {return H5Tcopy(H5T_NATIVE_INT);}
179  if(std::is_same<DataType, long>::value) {return H5Tcopy(H5T_NATIVE_LONG);}
180  if(std::is_same<DataType, double>::value) {return H5Tcopy(H5T_NATIVE_DOUBLE);}
181  if(std::is_same<DataType, float>::value) {return H5Tcopy(H5T_NATIVE_FLOAT);}
182  if constexpr (tc::has_member_scalar <DataType>::value) {return get_DataType<typename DataType::Scalar>();}
183  if constexpr (tc::has_member_value_type<DataType>::value){return get_DataType<typename DataType::value_type>();}
184  std::cout << "get_DataType can't match the type provided" << std::endl;
185  exit(1);
186  }

189  template<typename DataType>
190  hid_t get_DataSpace(const DataType &data){
191  if constexpr (tc::is_tensor<DataType>::value){
192  int rank = data.rank();
193  hsize_t dims[rank];
194  std::copy(data.dimensions().begin(), data.dimensions().end(), dims);
195  return H5Screate_simple(rank, dims, nullptr);
196  }
197  if constexpr (std::is_arithmetic<DataType>::value){
198  int rank = 1;
199  hsize_t dims[rank] = {1};
200  return H5Screate_simple(rank, dims, nullptr);
201  }
202  if constexpr (tc::is_matrix <DataType>::value) {
203  int rank = 2;
204  hsize_t dims[rank] = { (hsize_t) data.rows(), (hsize_t) data.cols() };
205  return H5Screate_simple(rank, dims, nullptr);
206  }
207  if constexpr(tc::is_instance_of<std::vector,DataType>::value){
208  int rank = 1;
209  hsize_t dims[rank] = {1};
210  return H5Screate_simple(rank, dims, nullptr);
211  }
212  std::cout << "get_DataSpace can't match the type provided" << std::endl;
213  exit(1);
214  }
