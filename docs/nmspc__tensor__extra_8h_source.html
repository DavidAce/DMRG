36  constexpr idxlistpair<long,N> idx (const T (&list1)[N], const T (&list2)[N]){
37  //Use numpy-style indexing for contraction. Each list contains a list of indices to be contracted for the respective
38  //tensors. This function zips them together into pairs as used in Eigen::Tensor module.
39  static_assert(std::is_integral_v<T>);
40  Eigen::array<Eigen::IndexPair<long>,N> pairlistOut;
41  for(unsigned long i = 0; i < N; i++){
42  pairlistOut[i] = Eigen::IndexPair<long>{list1[i], list2[i]};
43  }
44  return pairlistOut;
45  }
