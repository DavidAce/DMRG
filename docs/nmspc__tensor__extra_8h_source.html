47  constexpr idxlistpair<long,N> idx (const T (&list1)[N], const T (&list2)[N]){
48  //Use numpy-style indexing for contraction. Each list contains a list of indices to be contracted for the respective
49  //tensors. This function zips them together into pairs as used in Eigen::Tensor module. This does not sort the indices in decreasing order.
50  static_assert(std::is_integral_v<T>);
51  Eigen::array<Eigen::IndexPair<long>,N> pairlistOut;
52  for(unsigned long i = 0; i < N; i++){
53  pairlistOut[i] = Eigen::IndexPair<long>{list1[i], list2[i]};
54  }
55  return pairlistOut;

69  template<std::size_t NB, std::size_t N>
70  constexpr idxlistpair<long,N> sortIdx (const Eigen::array<long,NB> &dimensions, const long (&idx_ctrct_A)[N],const long (&idx_ctrct_B)[N]){
71  //When doing contractions, some indices may be larger than others. For performance, you want to
72  // contract the largest indices first. This will return a sorted index list in decreasing order.
73  Eigen::array<idx_dim_pair<long>,N> idx_dim_pair_list;
74  for (unsigned long i = 0; i < N; i++ ){
75  idx_dim_pair_list[i] = {idx_ctrct_A[i], idx_ctrct_B[i], dimensions[idx_ctrct_B[i]]};
76  }
77  std::sort(idx_dim_pair_list.begin(), idx_dim_pair_list.end(), [](const auto& i, const auto& j) { return i.dimB > j.dimB; } );
78  idxlistpair<long,N> pairlistOut;
79  for(unsigned long i = 0; i< N; i++){
80  pairlistOut[i] = Eigen::IndexPair<long>{idx_dim_pair_list[i].idxA, idx_dim_pair_list[i].idxB};
81  }
82  return pairlistOut;

298  template <typename Derived>
299  auto subtract_phase(Eigen::MatrixBase<Derived> &v){
300  using Scalar = typename Derived::Scalar;
301  std::vector<double> angles;
302  if constexpr (std::is_same<Scalar, std::complex<double>>::value) {
303  for (int i = 0; i < v.cols(); i++){
304  if (v.col(i)(0).imag() == 0.0){
305  angles.emplace_back(0.0);
306  continue;
307  }
308  angles.emplace_back(std::arg(v.col(i)(0)));
309  Scalar inv_phase = Scalar(0.0,-1.0) * angles.back();
310  Scalar exp_inv_phase = std::exp(inv_phase);
311 // std::cout << std::setprecision(20) << std::fixed << "inv_phase : " << inv_phase << std::endl;
312 // std::cout << std::setprecision(20) << std::fixed << "exp_inv_phase: " << exp_inv_phase << std::endl;
313  v.col(i) *= exp_inv_phase;
314  v.col(i) = (v.col(i).array().imag().cwiseAbs() > 1e-15 ).select(v.col(i), v.col(i).real());
315  }
316  }
317  return angles;
318  }
