119 template<typename Scalar>
120 void DenseMatrixProduct<Scalar>::FactorOP()
121 
122 /* Partial pivot LU decomposition
123  * Factors P(A-sigma*I) = LU
124  */
125 {
126  if(readyFactorOp){return;}
127  std::cout << "Starting LU \n";
128 
129  t_factorOp.tic();
130  assert(readyShift and "Shift value sigma has not been set.");
131  Scalar sigma;
132  if constexpr(std::is_same<Scalar,double>::value)
133  {
134  sigma = sigmaR;
135  lu.compute(A_matrix - sigmaR * Eigen::MatrixXd::Identity(L,L));
136  }
137  else
138  {
139  sigma = std::complex<double>(sigmaR,sigmaI);
140  lu.compute(A_matrix - sigma * Eigen::MatrixXd::Identity(L,L));
141  }
142 
143  readyFactorOp = true;
144  t_factorOp.toc();
145  std::cout << "Finished LU \n";
146  std::cout << "Time LU Op [ms]: " << std::fixed << std::setprecision(3) << t_factorOp.get_last_time_interval() * 1000 <<'\n';
147 
148 }

153 template<typename Scalar>
154 void DenseMatrixProduct<Scalar>::MultOPv(Scalar* x_in_ptr, Scalar* x_out_ptr) {
155  using namespace eigutils::eigSetting;
156  assert(readyFactorOp and "FactorOp() has not been run yet.");
157  switch (side){
158  case Side::R: {
159  Eigen::Map<VectorType> x_in (x_in_ptr,L);
160  Eigen::Map<VectorType> x_out (x_out_ptr,L);
161  x_out.noalias() = lu.solve(x_in);
162  break;
163  }
164  case Side::L: {
165  Eigen::Map<VectorTypeT> x_in (x_in_ptr,L);
166  Eigen::Map<VectorTypeT> x_out (x_out_ptr,L);
167  x_out.noalias() = x_in * lu.inverse();
168  break;
169  }
170  }
171  counter++;
172 }

177 template<typename Scalar>
178 void DenseMatrixProduct<Scalar>::MultAx(Scalar* x_in, Scalar* x_out) {
179  using namespace eigutils::eigSetting;
180  switch (form){
181  case Form::NONSYMMETRIC:
182  switch (side) {
183  case Side::R: {
184  Eigen::Map<VectorType> x_vec_in (x_in, L);
185  Eigen::Map<VectorType> x_vec_out(x_out, L);
186  x_vec_out.noalias() = A_matrix * x_vec_in ;
187  break;
188  }
189  case Side::L: {
190  Eigen::Map<VectorTypeT> x_vec_in(x_in, L);
191  Eigen::Map<VectorTypeT> x_vec_out(x_out, L);
192  x_vec_out.noalias() = x_vec_in * A_matrix;
193  break;
194  }
195  }
196  break;
197  case Form::SYMMETRIC: {
198  Eigen::Map<VectorType> x_vec_in(x_in, L);
199  Eigen::Map<VectorType> x_vec_out(x_out, L);
200  x_vec_out.noalias() = A_matrix.template selfadjointView<Eigen::Upper>() * x_vec_in;
201  break;
202  }
203  }
204  counter++;
205 }
