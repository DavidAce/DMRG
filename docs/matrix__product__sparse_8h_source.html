122 template<typename Scalar>
123 void SparseMatrixProduct<Scalar>::FactorOP()
124 
125 /* Sparse decomposition
126  * Factors P(A-sigma*I) = LU
127  */
128 
129 { if(readyFactorOp){return;}
130  assert(readyShift and "Shift value sigma has not been set.");
131  Scalar sigma;
132  t_factorOp.tic();
133  if constexpr(std::is_same<Scalar,double>::value)
134  {
135  sigma = sigmaR;
136  lu_dense.compute(A_matrix - sigmaR * Eigen::MatrixXd::Identity(L,L));
137  }
138  else
139  {
140  sigma = std::complex<double>(sigmaR,sigmaI);
141  lu_dense.compute(A_matrix - sigma * Eigen::MatrixXd::Identity(L,L));
142  }
143 
144  sparseLU = false;
145  t_factorOp.toc();
146  readyFactorOp = true;
147  std::cout << "Time Factor Op [ms]: " << std::fixed << std::setprecision(3) << t_factorOp.get_last_time_interval() * 1000 <<'\n';
148 }

150 template<typename Scalar>
151 void SparseMatrixProduct<Scalar>::MultOPv(Scalar* x_in_ptr, Scalar* x_out_ptr) {
152  using namespace eigutils::eigSetting;
153  assert(readyFactorOp and "FactorOp() has not been run yet.");
154  using VectorType = Eigen::Matrix<Scalar,Eigen::Dynamic,1>;
155  Eigen::Map<VectorType> x_in (x_in_ptr,L);
156  Eigen::Map<VectorType> x_out (x_out_ptr,L);
157  switch (side){
158  case Side::R: {
159  if(sparseLU){x_out.noalias() = lu_sparse.solve(x_in);}
160  else {x_out.noalias() = lu_dense.solve(x_in);}
161  break;
162  }
163  case Side::L: {
164  std::cerr << "Left sided sparse shift invert hasn't been implemented yet..." << std::endl;
165  exit(1);
166  break;
167  }
168  }
169  counter++;
170 }

174  using namespace eigutils::eigSetting;
175  switch (form){

179  Eigen::Map<VectorType> x_vec_in (x_in, L);

181  x_vec_out.noalias() = A_matrix * x_vec_in ;
182 
183  break;
184  }
185  case Side::L: {
186  Eigen::Map<VectorTypeT> x_vec_in(x_in, L);
187  Eigen::Map<VectorTypeT> x_vec_out(x_out, L);
188  x_vec_out.noalias() = x_vec_in * A_matrix;
189  break;
190  }
191  }
192  break;
193  case Form::SYMMETRIC: {
194  Eigen::Map<VectorType> x_vec_in (x_in, L);
195  Eigen::Map<VectorType> x_vec_out(x_out, L);

197  break;

199  }

201 }
