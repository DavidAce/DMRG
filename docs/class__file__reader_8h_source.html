57  T find_parameter(std::string param_requested){
58  if (file.is_open()) {
59  file.clear();
60  file.seekg(0, std::ios::beg);
61 
62 
63  std::string param_key;
64  std::string param_val;
65  std::string line;
66 
67  while (!file.eof()) {
68  getline(file, line);
69  if(!is_parameterline(line)){continue;}
70  std::istringstream is(line);
71  getline(is,param_key, '=');
72  getline(is,param_val, '\n');
73  param_val = param_val.substr(0, find_comment_character(param_val));
74  remove_spaces(param_requested);
75  remove_spaces(param_key);
76  remove_spaces(param_val);
77 
78  std::transform(param_key.begin(), param_key.end(), param_key.begin(), ::tolower);
79  std::transform(param_requested.begin(), param_requested.end(), param_requested.begin(), ::tolower);
80  if (param_requested == param_key && !param_key.empty()) {
81  ccout(2) << "Loading line: " << line << std::endl;
82  if constexpr (std::is_same<T,int>::value){
83  if (has_only_digits(param_val)) {
84  try {
85  T parsed_val = std::stoi(param_val);
86  return parsed_val;
87  }
88  catch (...) {std::cerr << "Error reading parameter from file: Unknown error." <<std::endl; }
89  }else{
90  std::cerr << "Error reading parameter from file. Wrong format: [" << param_val << "]. Expected an integer."<< std::endl;
91  }
92  }
93  if constexpr (std::is_same<T,long>::value){
94  if (has_only_digits(param_val)) {
95  try {
96  T parsed_val = std::stol(param_val);
97  return parsed_val;
98  }
99  catch (...) {std::cerr << "Error reading parameter from file: Unknown error." << std::endl; }
100  }else{
101  std::cerr << "Error reading parameter from file. Wrong format: [" << param_val << "]. Expected a long integer."<< std::endl;
102  }
103  }
104 
105  if constexpr (std::is_same<T,double>::value){
106  try {
107  return std::stod(param_val);
108  }
109  catch (...) {
110  std::cerr << "Error reading parameter from file: Unknown error: [" << param_val << "]. Expected a double." << std::endl;
111  }
112  }
113  if constexpr (std::is_same<T,bool>::value) {
114  if (param_val == "true") { return true; }
115  if (param_val == "false") { return false;}
116  }
117 
118  if constexpr (std::is_same<T,std::string>::value){
119  return param_val;
120  }
121 
122  std::cerr << "\nCritical error when reading parameter from file. Possible type mismatch." << std::endl;
123  std::cerr << "Requested : [" << param_requested << "]" << " with type: [" << typeid(T).name() << "]"<< std::endl;
124  std::cerr << "Found key : [" << param_key << "]" << std::endl;
125  std::cerr << "Found val : [" << param_val << "]" << std::endl;
126  std::cerr << "Exiting..." << std::endl << std::flush;
127  exit(1);
128  }
129  }
130  std::cerr << "Input file does not contain a parameter matching your query: [" << param_requested << "]" << std::endl;
131  std::cerr << "Exiting..." << std::endl << std::flush;
132  exit(1);
133  }
134  else {

136  exit(1);
137  }
