154  template<typename DataType>
155  constexpr hid_t get_DataType() const {
156  if constexpr (std::is_same<DataType, int>::value) {return H5Tcopy(H5T_NATIVE_INT);}
157  if constexpr (std::is_same<DataType, long>::value) {return H5Tcopy(H5T_NATIVE_LONG);}
158  if constexpr (std::is_same<DataType, unsigned int>::value) {return H5Tcopy(H5T_NATIVE_UINT);}
159  if constexpr (std::is_same<DataType, unsigned long>::value) {return H5Tcopy(H5T_NATIVE_ULONG);}
160  if constexpr (std::is_same<DataType, double>::value) {return H5Tcopy(H5T_NATIVE_DOUBLE);}
161  if constexpr (std::is_same<DataType, float>::value) {return H5Tcopy(H5T_NATIVE_FLOAT);}
162  if constexpr (std::is_same<DataType, std::complex<double>>::value){return H5Tcopy(H5T_COMPLEX_DOUBLE);}
163  if constexpr (std::is_same<DataType, H5T_COMPLEX_STRUCT>::value) {return H5Tcopy(H5T_COMPLEX_DOUBLE);}
164  if constexpr (std::is_same<DataType, char>::value) {return H5Tcopy(H5T_C_S1);}
165  if constexpr (std::is_same<DataType, std::string>::value) {return H5Tcopy(H5T_C_S1);}
166  if constexpr (tc::is_eigen_tensor<DataType>::value) {return get_DataType<typename DataType::Scalar>();}
167  if constexpr (tc::is_eigen_matrix<DataType>::value) {return get_DataType<typename DataType::Scalar>();}
168  if constexpr (tc::is_eigen_array<DataType>::value) {return get_DataType<typename DataType::Scalar>();}
169  if constexpr (tc::is_vector<DataType>::value) {return get_DataType<typename DataType::value_type>();}
170  if constexpr (tc::has_member_scalar <DataType>::value) {return get_DataType<typename DataType::Scalar>();}
171  if constexpr (tc::has_member_value_type <DataType>::value) {return get_DataType<typename DataType::value_type>();}
172 
173  std::cerr << "get_DataType could not match the type provided" << std::endl;
174  exit(1);
175  }

181  if constexpr (tc::has_member_size<DataType>::value) {return data.size();} //Fails on clang?
182  else if constexpr (std::is_arithmetic<DataType>::value) {return 1;}

185  else if constexpr (tc::is_eigen_tensor<DataType>::value) {return data.size();}

243  return dims;
244  }

302 std::vector<class_hdf5_file::H5T_COMPLEX_STRUCT> class_hdf5_file::convert_complex_data(const DataType &data){
303 
304  if constexpr(tc::is_eigen_tensor<DataType>::value or tc::is_eigen_matrix_or_array<DataType>()){
305  if constexpr(std::is_same<typename DataType::Scalar, std::complex<double>>::value) {
306  std::vector<H5T_COMPLEX_STRUCT> new_data(data.size());
307  for (int i = 0; i < data.size(); i++) {
308  new_data[i].real = data(i).real();
309  new_data[i].imag = data(i).imag();
310  }
311  return new_data;
312  }
313  }
314  if constexpr(tc::is_vector<DataType>::value) {
315  if constexpr(std::is_same<typename DataType::value_type, std::complex<double>>::value) {
316  std::vector<H5T_COMPLEX_STRUCT> new_data(data.size());
317  for (int i = 0; i < data.size(); i++) {
318  new_data[i].real = data[i].real();
319  new_data[i].imag = data[i].imag();
320  }
321  return new_data;
322  }
323  }
324 
325  if constexpr(std::is_arithmetic<DataType>::value and std::is_same<DataType, std::complex<double>>::value){
326  std::vector<H5T_COMPLEX_STRUCT> new_data(1);
327  new_data[0].real = data.real();
328  new_data[0].imag = data.imag();
329  return new_data;
330  }
331  //This should never happen, but is here so that we compile successfully.
332  assert(NAN == NAN and "Big error! Tried to convert non-complex data_struct to complex");
333  return std::vector<H5T_COMPLEX_STRUCT>();
334 }
